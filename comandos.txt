#compilar
sudo docker build -t webserver  .

#listar imagenes
sudo docker images

#ver imagenes en ejecución
sudo docker ps

#ver imagenes en ejecución con detenidas
sudo docker ps -a

#borrar imagen y contenedores 
sudo docker rmi webserver -f

#borrar container
sudo docker rm webserver-container

#ver detalles del container
sudo docker inspect webserver-container

#correr shell dentro del contenedores
sudo docker exec -it webserver-container bash

una vez dentro instalar el ps para ver los procesos en ejecución
apt-get update
apt-get install -y procps

ver procesos corriendo
ps aux

#tag image
sudo docker tag jmmarsan.webserver jmmarsan/webserver

#push image to dockerhub
docker push jmmarsan/webserver
